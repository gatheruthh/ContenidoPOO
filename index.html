Código Completo: Lógica Secuencial

SIN CLASES

Concepto: Programación basada en el flujo de ejecución directo. Se observa el uso de métodos estáticos que actúan como "subrutinas".

internal class Program {
    static void Main(string[] args) {
        string nombreTrabajador;
        int opcion;
        float sueldoBase, isr, sueldoBruto = 0, pagoHora;
        int horasT;

        Console.WriteLine("1. Administrativo");
        Console.WriteLine("2. Profesor de asignatura");
        Console.WriteLine("3. Profesor de Tiempo Completo");
        Console.WriteLine("Seleccione el tipo de empleado a calcular su nómina");
        opcion = int.Parse(Console.ReadLine());

        Console.WriteLine("Nombre del trabajador:");
        nombreTrabajador = Console.ReadLine();

        switch (opcion) {
            case 1:
                Console.WriteLine("Ingresa su sueldo base:");
                sueldoBase = float.Parse(Console.ReadLine());
                Console.WriteLine("Ingresa el impuesto sobre la renta (ISR):");
                isr = float.Parse(Console.ReadLine());
                sueldoBruto = CalcularNominaAdmon(sueldoBase, isr);
                break;
            case 2:
                Console.WriteLine("Ingresa las horas trabajadas:");
                pagoHora = float.Parse(Console.ReadLine());
                Console.WriteLine("Ingresa el pago por hora:");
                horasT = int.Parse(Console.ReadLine());
                Console.WriteLine("Ingresa el impuesto sobre la renta (ISR):");
                isr = float.Parse(Console.ReadLine());
                sueldoBruto = CalcularNominaAsignatura(pagoHora, horasT, isr);
                break;
            case 3:
                Console.WriteLine("Ingresa el pago por hora:");
                pagoHora = float.Parse(Console.ReadLine());
                Console.WriteLine("Ingresa el impuesto sobre la renta (ISR):");
                isr = float.Parse(Console.ReadLine());
                sueldoBruto = CalcularNominaCompleto(pagoHora, isr);
                break;
        }

        Console.WriteLine(nombreTrabajador + " Tu sueldo es: " + sueldoBruto);
        Console.ReadKey();
    }

    // MÉTODOS ESTÁTICOS (Lógica desconectada de los datos)
    static float CalcularNominaAdmon(float sueldoBase, float isr) {
        return sueldoBase - (sueldoBase * isr / 100);
    }

    static float CalcularNominaAsignatura(float pagoHora, int horasT, float isr) {
        return (pagoHora * horasT) - isr;
    }

    static float CalcularNominaCompleto(float pagoHora, float isr) {
        return (pagoHora * 40) - isr;
    }
}

Código Completo: Transición a Clases

INSTANCIACIÓN

Saber Hacer: "Construir objetos a partir de una clase". Aquí se separan las responsabilidades en una clase externa.

// DEFINICIÓN DE LA CLASE
internal class clsAdministrativos {
    // Método de clase
    public float CalcularNominaAdmon(float sueldoBase, float isr) {
        return sueldoBase - (sueldoBase * isr / 100);
    }
}

// PROGRAMA PRINCIPAL
internal class Program {
    static void Main(string[] args) {
        float sb, i, res;
        Console.WriteLine("Sueldo Base:");
        sb = float.Parse(Console.ReadLine());
        Console.WriteLine("ISR:");
        i = float.Parse(Console.ReadLine());

        // PASO CRÍTICO: Instanciación del objeto
        clsAdministrativos admon = new clsAdministrativos();
        
        // Uso del objeto para realizar el cálculo
        res = admon.CalcularNominaAdmon(sb, i);
        
        Console.WriteLine("Resultado: " + res);
    }
}

Código Completo: Abstracción de Atributos

ESTADO Y COMPORTAMIENTO

Concepto: El objeto ahora mantiene su propio Estado (Atributos) y no requiere pasar datos por parámetros en sus métodos.

internal class clsAdministrativos {
    // ATRIBUTOS (Estado)
    public string nombreTrabajador;
    public float isr;
    public float sueldoBase;
    public float sueldoNeto;

    // MÉTODO (Comportamiento que usa el estado interno)
    public float CalcularNomina() {
        sueldoNeto = sueldoBase - (sueldoBase * isr / 100);
        return sueldoNeto;
    }
}

internal class Program {
    static void Main(string[] args) {
        clsAdministrativos admon = new clsAdministrativos();
        
        Console.WriteLine("Nombre:");
        admon.nombreTrabajador = Console.ReadLine();
        Console.WriteLine("Sueldo Base:");
        admon.sueldoBase = float.Parse(Console.ReadLine());
        Console.WriteLine("ISR:");
        admon.isr = float.Parse(Console.ReadLine());

        // El objeto sabe cómo calcularse a sí mismo
        float resultado = admon.CalcularNomina();
        Console.WriteLine(admon.nombreTrabajador + " gana neto: " + resultado);
    }
}

Código Completo: Arquitectura Final (Herencia)

OPTIMIZACIÓN

Guía Temática: "Realizar la optimización por medio de herencia y polimorfismo". Código completo del sistema jerárquico.

// CLASE BASE (ABSTRACTA)
internal abstract class clsTrabajadores {
    public string nombreTrabajador;
    public float isr;
    public abstract float CalcularNomina(); 
}

// ESPECIALIZACIONES
internal class clsAdministrativos : clsTrabajadores {
    public float SueldoBase;
    public override float CalcularNomina() => SueldoBase - (SueldoBase * isr / 100);
}

internal class clsProfesorAsignatura : clsTrabajadores {
    public float PagoHora;
    public int HorasT;
    public override float CalcularNomina() => (PagoHora * HorasT) - isr;
}

internal class clsProfesorTiempoCompleto : clsTrabajadores {
    public float PagoHora;
    public override float CalcularNomina() => (PagoHora * 40) - isr;
}
// PROGRAMA DE CONTROL (Switch Completo con Objetos)
static void Main(string[] args) {
    int op;
    clsAdministrativos admon = new clsAdministrativos();
    clsProfesorAsignatura PA = new clsProfesorAsignatura();
    clsProfesorTiempoCompleto PTC = new clsProfesorTiempoCompleto();

    Console.WriteLine("1. Administrativo\n2. Asignatura\n3. Tiempo Completo\nSeleccione:");
    op = int.Parse(Console.ReadLine());

    switch (op) {
        case 1:
            Console.WriteLine("Nombre:");
            admon.nombreTrabajador = Console.ReadLine();
            Console.WriteLine("Sueldo Base:");
            admon.SueldoBase = float.Parse(Console.ReadLine());
            admon.isr = float.Parse(Console.ReadLine());
            Console.WriteLine(admon.nombreTrabajador + " Sueldo: " + admon.CalcularNomina());
            break;
        case 2:
            Console.WriteLine("Nombre:");
            PA.nombreTrabajador = Console.ReadLine();
            Console.WriteLine("Pago Hora:");
            PA.PagoHora = float.Parse(Console.ReadLine());
            Console.WriteLine("Horas:");
            PA.HorasT = int.Parse(Console.ReadLine());
            PA.isr = float.Parse(Console.ReadLine());
            Console.WriteLine(PA.nombreTrabajador + " Sueldo: " + PA.CalcularNomina());
            break;
        case 3:
            Console.WriteLine("Nombre:");
            PTC.nombreTrabajador = Console.ReadLine();
            Console.WriteLine("Pago Hora:");
            PTC.PagoHora = float.Parse(Console.ReadLine());
            PTC.isr = float.Parse(Console.ReadLine());
            Console.WriteLine(PTC.nombreTrabajador + " Sueldo: " + PTC.CalcularNomina());
            break;
    }
}
